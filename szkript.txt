DROP TABLE Kozlekedik;
DROP TABLE Allomas;
DROP TABLE Jarat;
DROP TABLE Jegy;
DROP TABLE Hir;
DROP TABLE Szerelveny;
CREATE TABLE Allomas (
    Nev VARCHAR(255) PRIMARY KEY,
    Varos VARCHAR(255)
);
CREATE TABLE Jarat (
    Vonatszam INT PRIMARY KEY,
    Honnan VARCHAR(255),
    Hova VARCHAR(255),
    Indulasi_ido TIMESTAMP,
    Menetido TIMESTAMP,
    Vonattipus VARCHAR(255)
);
CREATE TABLE Kozlekedik (
    AllomasNev VARCHAR(255),
    Vonatszam INTEGER,
    Erkezes TIMESTAMP, 
    Indulas TIMESTAMP,
    FOREIGN KEY (AllomasNev) REFERENCES Allomas(Nev),
    FOREIGN KEY (Vonatszam) REFERENCES Jarat(Vonatszam)
);
CREATE TABLE Felhasznalo (
    Email VARCHAR(255) PRIMARY KEY,
    Nev VARCHAR(255),
    Eletkor INT,
    Kedvezmenytipus VARCHAR(255),
    Igazolvanyszam VARCHAR(255),
    Profilkep BLOB,
    Vonatszam INT,
    FOREIGN KEY (Vonatszam) REFERENCES Jarat(Vonatszam)
);
CREATE TABLE Jegy (
    Tipus VARCHAR(255) PRIMARY KEY,
    Ar INT,
    Feltetel CLOB,
    Idotartam INT
);
CREATE TABLE Vasarol (
    Email VARCHAR(255),
    Tipus VARCHAR(255),
    FOREIGN KEY (Email) REFERENCES Felhasznalo(Email),
    FOREIGN KEY (Tipus) REFERENCES Jegy(Tipus)
);
CREATE TABLE Hir (
    HirId INT PRIMARY KEY,
    Cim VARCHAR(255),
    Szoveg CLOB,
    Datum DATE
);
CREATE TABLE Szerelveny (
    SzNev VARCHAR(255) PRIMARY KEY,
    Gyartasi_ev INT,
    Meghajtas VARCHAR(255),
    Kapacitas INT,
    Kerekparhelyek_szama INT,
    Osztaly INT 
);
