-- Táblák törlése, ha véletlenül lennének (hibát elv. nem okoznak)

DROP TABLE Vasarol;
DROP TABLE Felhasznalo;
DROP TABLE Kozlekedik;
DROP TABLE Allomas;
DROP TABLE Jarat;
DROP TABLE Jegy;
DROP TABLE Hir;
DROP TABLE Szerelveny;

-- Táblák létrehozása

CREATE TABLE Allomas (
    Nev VARCHAR(255) PRIMARY KEY,
    Varos VARCHAR(255)
);
CREATE TABLE Jarat (
    Vonatszam INT PRIMARY KEY,
    Honnan VARCHAR(255),
    Hova VARCHAR(255),
    Indulasi_ido TIMESTAMP,
    Menetido TIMESTAMP,
    Vonattipus VARCHAR(255)
);
CREATE TABLE Kozlekedik (
    AllomasNev VARCHAR(255),
    Vonatszam INTEGER,
    Erkezes TIMESTAMP, 
    Indulas TIMESTAMP,
    FOREIGN KEY (AllomasNev) REFERENCES Allomas(Nev) ON DELETE CASCADE,
    FOREIGN KEY (Vonatszam) REFERENCES Jarat(Vonatszam) ON DELETE CASCADE
);
CREATE TABLE Felhasznalo (
    Email VARCHAR(255) PRIMARY KEY,
    Jelszo VARCHAR(255),
    Nev VARCHAR(255),
    Eletkor INT,
    Kedvezmenytipus VARCHAR(255),
    Igazolvanyszam VARCHAR(255),
    Profilkep BLOB,
    Admin NUMBER(0,1)
    
    
);
CREATE TABLE Jegy (
    Tipus VARCHAR(255) PRIMARY KEY,
    Ar INT,
    Feltetel CLOB,
    Idotartam INT
);

CREATE TABLE Vasarol (
    id INT GENERATED by default on null as IDENTITY,
    Email VARCHAR(255),
    Tipus VARCHAR(255),
    Kezdet DATE,
    FOREIGN KEY (Email) REFERENCES Felhasznalo(Email) ON DELETE CASCADE,
    FOREIGN KEY (Tipus) REFERENCES Jegy(Tipus) ON DELETE CASCADE
);

CREATE TABLE Hir (
    HirId INT PRIMARY KEY,
    Cim VARCHAR(255),
    Szoveg CLOB,
    Datum DATE
);
CREATE TABLE Szerelveny (
    SzNev VARCHAR(255) PRIMARY KEY,
    Gyartasi_ev INT,
    Meghajtas VARCHAR(255),
    Kapacitas INT,
    Kerekparhelyek_szama INT,
    Osztaly INT 
);
